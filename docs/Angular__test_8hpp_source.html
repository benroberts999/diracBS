71 double UnitTest::helper::ck_loop(const Angular::Ck_ab &Ck,
72  Angular::Ck_ab &Ck_m) {
73  const auto max2j = Ck.max_tj();
74  double max = 0.0;
75 
76  for (int kia = 0;; ++kia) {
77  const auto tja = Angular::twojFromIndex(kia);
78  if (tja > max2j)
79  break;
80  const auto ka = Angular::kappaFromIndex(kia);
81  for (int kib = 0;; ++kib) {
82  const auto tjb = Angular::twojFromIndex(kib);
83  if (tjb > max2j)
84  break;
85  const auto kb = Angular::kappaFromIndex(kib);
86 
87  // loop through all k multipolarities:
88  for (int k = 0; k <= max2j; ++k) {
89 
90  const auto c1 = Ck.get_Ckab(k, ka, kb);
91  const auto c1_m = Ck_m.get_Ckab_mutable(k, ka, kb);
92  const auto c2 = Angular::Ck_kk(k, ka, kb);
93  const auto sign = ((tja + 1) / 2) % 2 == 0 ? 1 : -1;
94  const auto c3 = Angular::parity(Angular::l_k(ka), Angular::l_k(kb), k) *
95  Angular::threej_2(tja, tjb, 2 * k, -1, 1, 0) * sign *
96  std::sqrt((tja + 1) * (tjb + 1));
97 
98  const auto tc1 = Ck.get_tildeCkab(k, ka, kb);
99  const auto tc2 =
100  Angular::parity(Angular::l_k(ka), Angular::l_k(kb), k) *
101  Angular::threej_2(tja, tjb, 2 * k, -1, 1, 0) *
102  std::sqrt((tja + 1) * (tjb + 1));
103 
104  const auto tj1 = Ck.get_3jkab(k, ka, kb);
105  const auto tj2 = Angular::threej_2(tja, tjb, 2 * k, -1, 1, 0);
106  const auto tj3 = gsl_sf_coupling_3j(tja, tjb, 2 * k, -1, 1, 0);
107 
108  const auto l1 = Ck.get_Lambdakab(k, ka, kb);
109  const auto l2 =
110  tj1 * tj1 * Angular::parity(Angular::l_k(ka), Angular::l_k(kb), k);
111 
112  max = qip::max_abs(max, c1 - c2, c1_m - c1, c2 - c3, tc1 - tc2,
113  tj1 - tj2, tj2 - tj3, l1 - l2);
114  }
115  }
116  }
117  return max;
118 }
