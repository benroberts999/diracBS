178  std::vector<double> break_points(std::size_t nbreaks) const {
179  // Breakpoints: (0, r0, ..., rmax)
180  // knots are the same, but have k-fold "multiplicity" at the ends
181  std::vector<double> breaks;
182 
183  // Set breakpoints according to grid (align exactly w/ grid points)
184  // Go 1 past beg/end, to ensure f_spl(r) is in spline interp. region
185  auto points =
186  qip::uniform_range(m_rmin_index - 1, m_rmax_index + 1, nbreaks - 1);
187  breaks.reserve(points.size());
188  breaks.push_back(0.0);
189  for (auto p : points) {
190  breaks.push_back(m_rgrid_ptr->r[p]);
191  }
192 
193  // Ensure there are at least k grid points between knots:
194  // First point may be less than k points from r0
195  std::adjacent_difference(points.begin(), points.end(), points.begin());
196  const auto min = *std::min_element(points.begin() + 2, points.end());
197  if (min <= m_order_k) {
198  std::cerr
199  << "\nWARNING 233 in BSplines: not enough points between knots for k="
200  << m_order_k << ", " << min << "\n";
201  }
202 
203  return breaks;
204  }

262  // p0 is first non-zero point
263  // pinf is first zero point (i.e., last non-zero point + 1)
264  m_ends.clear();
265  for (const auto &Bk : m_Bk) {
266  auto is_nonzero = [](auto a) { return std::abs(a) > 0.0; };
267  auto first_nonzero = std::find_if(Bk.begin(), Bk.end(), is_nonzero);
268  auto first_zero = std::find_if_not(first_nonzero, Bk.end(), is_nonzero);
269  auto p0 =
270  static_cast<std::size_t>(std::distance(Bk.begin(), first_nonzero));
271  auto pinf =
272  static_cast<std::size_t>(std::distance(Bk.begin(), first_zero));
273  m_ends.emplace_back(p0, pinf);
274  }
