55  bool pass = true;

57  { // First, test quadrature integration method:
58 
59  // Define a wavefunction-like function, func,
60  auto func = [](double x) {
61  return x * std::exp(-0.2 * x) * (1.0 + 2.0 * std::sin(x));
62  };
63  // that has an exact integral, Intfunc = \int_0^x(func):
64  auto Intfunc = [](double x) {
65  return -(5.0 / 169.0) * std::exp(-0.2 * x) *
66  (169.0 * (5.0 + x) + 5.0 * (5.0 + 13.0 * x) * std::cos(x) +
67  (13.0 * x - 60.0) * std::sin(x));
68  };
69 
70  // Perform the numerical integrals using a different grids, compare to exact
71  const auto pts_lst = std::vector<std::size_t>{1000};
72  for (const auto pts : pts_lst) {
73  const Grid grll(1.0e-6, 100.0, pts, GridType::loglinear, 10);
74  const Grid grlog(1.0e-6, 100.0, pts, GridType::logarithmic, 0);
75 
76  std::vector<double> vll, vlog;
77  for (const auto &r : grll.r)
78  vll.push_back(func(r));
79  for (const auto &r : grlog.r)
80  vlog.push_back(func(r));
81 
82  // numerical integration (on grid):
83  const auto intll = NumCalc::integrate(grll.du, 0, 0, vll, grll.drdu);
84  const auto intlog = NumCalc::integrate(grlog.du, 0, 0, vlog, grlog.drdu);
85 
86  // Account for possibility r0, rmax slightly different:
87  const auto exactll = Intfunc(grll.r.back()) - Intfunc(grll.r.front());
88  const auto exactlog = Intfunc(grlog.r.back()) - Intfunc(grlog.r.front());
89 
90  pass &=

92  "Quad[" + std::to_string(NumCalc::Nquad) +
93  "] int (log-lin) b=10 N=" + std::to_string(pts),
94  (intll - exactll) / exactll, 0.0, 1.0e-13);
95  pass &= qip::check_value(
96  &obuff,
97  "Quad[" + std::to_string(NumCalc::Nquad) +
98  "] int (logarithmic) N=" + std::to_string(pts),
99  (intlog - exactlog) / exactlog, 0.0, 1.0e-4 * (500.0 / double(pts)));

173  const auto maxtj = std::max_element(wf.basis.cbegin(), wf.basis.cend(),
174  DiracSpinor::comp_j)
175  ->twoj();

187

191

201  Coulomb::Rk_abcd(Fa, Fb, Fc, Fd, k);

205  // This k_min should have correct parity rule too

208  : 0.0;
209

211  std::abs(qip::max_difference(Q1, Q2, Q3, Q4, Q5));
212  if (delQ > worstQ)

222  if (Angular::Ck_kk_SR(l, Fa.k, Fd.k) &&

225  P4 += (2 * k + 1) *
226  sj.get_6j(Fa.twoj(), Fc.twoj(), Fb.twoj(), Fd.twoj(), k,

229  }
230  }
231

233  const auto [kminP, kmaxP] = Coulomb::k_minmax_P(Fa, Fb, Fc, Fd);
234  // This k_min CANNOT contain correct parity rule
235  const auto P5 =
236  (k >= kminP && k <= kmaxP) ? Yij.Pk(k, Fa, Fb, Fc, Fd) : 0.0;
237 
238  const auto delP = std::abs(qip::max_difference(P1, P2, P4, P5));
239  if (delP > worstP)
240  worstP = delP;
241 
242  // calc W
243  const auto W1 = Q1 + P1;
244  const auto W2 = Coulomb::Wk_abcd(Fa, Fb, Fc, Fd, k);
245 
246  // test the 'Wk' version, including k_minmax_P
247  const auto [kmin, kmax] = Coulomb::k_minmax_W(Fa, Fb, Fc, Fd);
248  // This k_min CANNOT contain correct parity rule
249  const auto W5 =
250  (k >= kmin && k <= kmax) ? Yij.Wk(k, Fa, Fb, Fc, Fd) : 0.0;
251 
252  const auto delW = std::abs(qip::max_difference(W1, W2, W5));

254  worstW = delW;
255 
256  } // k
257  } // d
258  } // c
259  } // b
260  } // a
261  pass &= qip::check_value(&obuff, "Qk_abcd ", worstQ, 0.0, 5.0e-13);
262  pass &= qip::check_value(&obuff, "Pk_abcd ", worstP, 0.0, 5.0e-14);
263  pass &= qip::check_value(&obuff, "Wk_abcd ", worstW, 0.0, 5.0e-14);
264  }

275 inline std::vector<double> UnitTest::helper::yk_naive(const DiracSpinor &Fa,
276  const DiracSpinor &Fb,
277  int k) {
278  const auto &gr = *Fa.rgrid;
279  std::vector<double> yk(gr.r.size());
280 #pragma omp parallel for
281  for (auto i = 0ul; i < yk.size(); ++i) {
282  auto r = gr.r[i];
283 
284  auto rtkr = [](double x, double y, int kk) {
285  return x < y ? std::pow(x / y, kk) / y : std::pow(y / x, kk) / x;
286  };
287 
288  std::vector<double> f;
289  f.reserve(gr.r.size());
290  for (auto j = 0ul; j < yk.size(); ++j) {
291  f.push_back(rtkr(r, gr.r[j], k) *
292  (Fa.f[j] * Fb.f[j] + Fa.g[j] * Fb.g[j]));
293  }
294 
295  const auto p0 = std::max(Fa.p0, Fb.p0);
296  const auto pi = std::min(Fa.pinf, Fb.pinf);
297  yk[i] = NumCalc::integrate(gr.du, p0, pi, f, gr.drdu);
298  }
299 
300  return yk;
301 }

304 inline double
305 UnitTest::helper::check_ykab_Tab(const std::vector<DiracSpinor> &a,
306  const std::vector<DiracSpinor> &b,
307  const Coulomb::YkTable &Yab) {
308  //
309  double worst = 0.0;
310  for (const auto &Fa : a) {
311  for (const auto &Fb : b) {
312  const auto [kmin, kmax] = Coulomb::k_minmax(Fa, Fb);
313  for (int k = kmin; k <= kmax; ++k) {
314  // Only check if Angular factor is non-zero (since Ykab only calc'd in
315  // this case)
316  if (!Angular::Ck_kk_SR(k, Fa.k, Fb.k))
317  continue;
318  const auto &y1 = Yab.get_yk_ab(k, Fa, Fb);
319  const auto y2 = Coulomb::yk_ab(Fa, Fb, k);
320  const auto y3 = Coulomb::yk_ab(Fb, Fa, k); // symmetric
321  const auto del = std::abs(qip::compare(y1, y2).first) +
322  std::abs(qip::compare(y2, y3).first);
323  if (del > worst)
324  worst = del;
325  }
326  }
327  }
328 
329  return worst;
330 }

333 inline std::vector<double>
334 UnitTest::helper::check_ykab(const std::vector<DiracSpinor> &orbs,
335  int max_del_n) {
336 
337  // Compared Yk as calculated by fast Coulomb::yk_ab routine (used in the code)
338  // to helper::yk_naive, a very slow, but simple version. In theory, should be
339  // exactly the same
340 
341  std::vector<double> worst; // = 0.0;
342  // nb: slow, so only check sub-set
343  for (auto ia = 0ul; ia < orbs.size(); ++ia) {
344  const auto &Fa = orbs[ia];
345  for (auto ib = ia; ib < orbs.size(); ++ib) {
346  const auto &Fb = orbs[ib];
347  const auto [kmin, kmax] = Coulomb::k_minmax(Fa, Fb);
348  for (int k = kmin; k <= kmax; ++k) {
349  if (!Angular::Ck_kk_SR(k, Fa.k, Fb.k))
350  continue;
351  if (std::abs(Fa.n - Fb.n) > max_del_n)
352  continue;
353  if (std::size_t(k + 1) > worst.size())
354  worst.resize(std::size_t(k + 1));
355  const auto y2 = Coulomb::yk_ab(Fa, Fb, k);
356  const auto y4 = helper::yk_naive(Fa, Fb, k); // slow
357 
358  const auto max = *std::max_element(cbegin(y4), cend(y4), qip::comp_abs);
359  const auto del = std::abs(qip::compare(y2, y4).first / max);
360 
361  if (del > worst[std::size_t(k)])
362  worst[std::size_t(k)] = del;
363  }
364  }
365  }
366 
367  return worst;
368 }

372 UnitTest::helper::check_Rkabcd(const std::vector<DiracSpinor> &orbs,
373  int max_del_n) {
374  double eps_R = 0.0;
375  const Coulomb::YkTable Yab(orbs.front().rgrid, &orbs);
376 #pragma omp parallel for
377  for (auto ia = 0ul; ia < orbs.size(); ia++) {
378  const auto &Fa = orbs[ia];
379  for (auto ib = 0ul; ib < orbs.size(); ib += 2) {
380  const auto &Fb = orbs[ib];
381  for (auto ic = ia; ic < orbs.size(); ic++) {
382  const auto &Fc = orbs[ic];
383  if (std::abs(Fa.n - Fc.n) > max_del_n)
384  continue;
385  if (std::abs(Fb.n - Fc.n) > max_del_n)
386  continue;
387  for (auto id = ib; id < orbs.size(); id += 2) {
388  const auto &Fd = orbs[id];
389  if (std::abs(Fb.n - Fd.n) > max_del_n)
390  continue;
391  const auto [kmin, kmax] = Coulomb::k_minmax(Fa, Fc);
392  for (int k = kmin; k <= kmax; ++k) {
393  if (!Angular::Ck_kk_SR(k, Fa.k, Fc.k))
394  continue;
395  if (!Angular::Ck_kk_SR(k, Fb.k, Fd.k))
396  continue;
397 
398  //---------
399  const auto &yac = Yab.get_yk_ab(k, Fa, Fc);
400  const auto &ybd = Yab.get_yk_ab(k, Fb, Fd);
401 
402  const auto r1a = Coulomb::Rk_abcd(Fa, Fb, Fc, Fd, k);
403  const auto r1b = Coulomb::Rk_abcd(Fb, Fa, Fd, Fc, k);
404  const auto r1c = Coulomb::Rk_abcd(Fc, Fd, Fa, Fb, k);
405  const auto r2a = Coulomb::Rk_abcd(Fa, Fc, ybd);
406  const auto r2b = Coulomb::Rk_abcd(Fb, Fd, yac);
407  const auto r2c = Coulomb::Rk_abcd(Fc, Fa, ybd);
408  const auto r3 = Fa * Coulomb::Rkv_bcd(Fa.k, Fb, Fc, Fd, k);
409  const auto r4 = Fa * Coulomb::Rkv_bcd(Fa.k, Fc, ybd);
410  const auto eps = std::max({r1a, r1b, r1c, r2a, r2b, r2c, r3, r4}) -
411  std::min({r1a, r1b, r1c, r2a, r2b, r2c, r3, r4});
412 #pragma omp critical(compare_epsR)
413  if (eps > eps_R) {
414  eps_R = eps;
415  }
416  }
417  }
418  }
419  }
420  }
421  return eps_R;
