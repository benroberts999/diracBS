56 inline auto rampedDamp(double a_beg, double a_end, int beg, int end) {
57  return [=](int i) {
58  if (i >= end)
59  return a_end;
60  if (i <= beg)
61  return a_beg;
62  return (a_end * (i - beg) + a_beg * (end - i)) / (end - beg);
