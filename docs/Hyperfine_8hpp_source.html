27 inline auto volotkaBW_F(double mu, double I_nuc, double l_pn, int gl)
28  -> Func_R2_R
29 // Function that returns generates + returns F_BW Bohr-Weiskopf
30 // gl = 1 for proton, =0 for neutron. Double allowed for testing..
31 // mu is in units of Nuclear Magneton!
32 {
33  const auto two_I = int(2 * I_nuc + 0.0001);
34  const auto two_l = int(2 * l_pn + 0.0001);
35  const auto g_l = double(gl); // just safety
36  const auto gI = mu / I_nuc;
37 
38  const auto K = (l_pn * (l_pn + 1.0) - (3. / 4.)) / (I_nuc * (I_nuc + 1.0));
39  const double g_s = (2.0 * gI - g_l * (K + 1.0)) / (1.0 - K);
40  std::cout << "BW using: gl=" << g_l << ", gs=" << g_s << ", l=" << l_pn
41  << ", gI=" << gI << " (I=" << I_nuc << ")\n";
42  const double factor =
43  (two_I == two_l + 1)
44  ? g_s * (1 - two_I) / (4.0 * (two_I + 2)) + g_l * 0.5 * (two_I - 1)
45  : g_s * (3 + two_I) / (4.0 * (two_I + 2)) +
46  g_l * 0.5 * two_I * (two_I + 3) / (two_I + 2);
47  if (two_I != two_l + 1 && two_I != two_l - 1) {
48  std::cerr << "\nFAIL:59 in Hyperfine (volotkaBW_F):\n "
49  "we must have I = l +/- 1/2, but we have: I,l = "
50  << I_nuc << "," << l_pn << "\n";
51  return [](double, double) { return 0.0; };
52  }
53  return [=](double r, double rN) {
54  return (r > rN) ? 1.0
55  : ((r * r * r) / (rN * rN * rN)) *
56  (1.0 - (3.0 / mu) * std::log(r / rN) * factor);
57  };

61 inline auto doublyOddBW_F(double mut, double It, double mu1, double I1,
62  double l1, int gl1, double I2, double l2) -> Func_R2_R
63 // F(r) * g = 0.5 [ g1F1 + g2F2 + (g1F1 - g2F2) * K]
64 // K = [I1(I1+1) - I2(I2+1)] / [I(I+1)]
65 // return F(r) [divide by g]
66 // volotkaBW_F(mu, I, l, g_l); //gl is 1 or 0
67 // g2 : from: g = 0.5 [ g1 + g2 + (g1 - g2) * K]
68 {
69  const auto K = (I1 * (I1 + 1.0) - I2 * (I2 + 1.0)) / (It * (It + 1.0));
70  const auto gt = mut / It;
71  const auto g1 = mu1 / I1;
72  const auto g2 = (g1 * (K + 1.0) - 2.0 * gt) / (K - 1.0);
73  const auto mu2 = g2 * I2;
74  const auto gl2 = (gl1 == 0) ? 1 : 0;
75  const auto F1 = volotkaBW_F(mu1, I1, l1, gl1);
76  const auto F2 = volotkaBW_F(mu2, I2, l2, gl2);
77  return [=](double r, double rN) {
78  return (0.5 / gt) * (g1 * F1(r, rN) + g2 * F2(r, rN) +
79  K * (g1 * F1(r, rN) - g2 * F2(r, rN)));
80  };
81 }
