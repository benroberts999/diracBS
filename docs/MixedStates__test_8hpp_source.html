21 std::pair<Case, Case> MS_loops(const Wavefunction &wf,
22  const DiracOperator::TensorOperator *h) {
23  // compare the best and worst! (of each operator)
24  Case best{999.0};
25  Case worst{0.0};
26 
27  const auto omega_mults = std::vector{0.0, 0.25};
28 
29 // for (const auto &Fv : wf.valence) {
30 #pragma omp parallel for
31  for (auto i = 0ul; i < wf.valence.size(); ++i) {
32  const auto &Fv = wf.valence[i];
33  const auto vl = wf.get_Vlocal(Fv.l());
34  for (const auto &Fm : wf.valence) {
35  if (Fm == Fv)
36  continue; // gives 0/0
37  if (h->isZero(Fv.k, Fm.k))
38  continue;
39 
40  if (Fm.k != Fv.k && std::abs(Fm.n - Fv.n) != 0)
41  continue;
42 
43  // std::cout << h->name() << " " << Fv.symbol() << " " << Fm.symbol()
44  // << "\n";
45 
46  const auto h_mv = h->reducedME(Fm, Fv);
47  // Don't check in cases where ME is extremely small. NOTE: This requires
48  // a good choice of units, since ME may be small due to small
49  // coeficient, which should not count as 'small'
50  if (std::abs(h_mv) < 1.0e-8) {
51  // std::cout << h_mv << "\n";
52  continue;
53  }
54 
55  auto hFv = h->reduced_rhs(Fm.k, Fv);
56  if (Fm.k == Fv.k) {
57  auto de = h->reducedME(Fv, Fv);
58  hFv -= de * Fv;
59  // std::cout << "de=" << de << "\n";
60  }
61 
62  for (const auto &w_mult : omega_mults) {
63  const auto w = std::abs(Fv.en * w_mult);
64 
65  const auto dFv =
66  HF::solveMixedState(Fm.k, Fv, w, vl, wf.alpha, wf.core, hFv);
67 
68  const auto lhs = Fm * dFv;
69  const auto rhs = h_mv / (Fv.en - Fm.en + w);
70  const auto eps = (lhs - rhs) / lhs;
71 
72 // find worst-case:
73 #pragma omp critical(find_max)
74  {
75  if (std::abs(eps) > std::abs(worst.eps)) {
76  worst.eps = eps;
77  worst.name = Fm.shortSymbol() + "|" + Fv.shortSymbol();
78  worst.w = w;
79  }
80  if (std::abs(eps) < std::abs(best.eps)) {
81  best.eps = eps;
82  best.name = Fm.shortSymbol() + "|" + Fv.shortSymbol();
83  best.w = w;
84  }
85  }
86  }
87  }
88  }
89  return {best, worst};
90 }
