235  static inline auto volotkaBW_F(double mu, double I_nuc, double l_pn, int gl)
236  -> Func_R2_R
237  // Function that returns generates + returns F_BW Bohr-Weiskopf
238  // gl = 1 for proton, =0 for neutron. Double allowed for testing..
239  // mu is in units of Nuclear Magneton!
240  {
241  const auto two_I = int(2 * I_nuc + 0.0001);
242  const auto two_l = int(2 * l_pn + 0.0001);
243  const auto g_l = double(gl); // just safety
244  const auto gI = mu / I_nuc;
245  const auto K = (l_pn * (l_pn + 1.0) - (3. / 4.)) / (I_nuc * (I_nuc + 1.0));
246  const double g_s = (2.0 * gI - g_l * (K + 1.0)) / (1.0 - K);
247  std::cout << "BW using: gl=" << g_l << ", gs=" << g_s << ", l=" << l_pn
248  << ", gI=" << gI << " (I=" << I_nuc << ")\n";
249  const double factor =
250  (two_I == two_l + 1)
251  ? g_s * (1 - two_I) / (4.0 * (two_I + 2)) + g_l * 0.5 * (two_I - 1)
252  : g_s * (3 + two_I) / (4.0 * (two_I + 2)) +
253  g_l * 0.5 * two_I * (two_I + 3) / (two_I + 2);
254  if (two_I != two_l + 1 && two_I != two_l - 1) {
255  std::cerr << "\nFAIL:59 in Hyperfine (volotkaBW_F):\n "
256  "we must have I = l +/- 1/2, but we have: I,l = "
257  << I_nuc << "," << l_pn << "\n";
258  return [](double, double) { return 0.0; };
259  }
260  return [=](double r, double rN) {
261  return (r > rN) ? 1. / (r * r)
262  : (r / (rN * rN * rN)) *
263  (1.0 - (3.0 / mu) * std::log(r / rN) * factor);
264  };
265  }

268  static inline auto doublyOddBW_F(double mut, double It, double mu1, double I1,
269  double l1, int gl1, double I2, double l2)
270  -> Func_R2_R
271  // F(r) * g = 0.5 [ g1F1 + g2F2 + (g1F1 - g2F2) * K]
272  // K = [I1(I1+1) - I2(I2+1)] / [I(I+1)]
273  // return F(r) [divide by g]
274  // volotkaBW_F(mu, I, l, g_l); //gl is 1 or 0
275  // g2 : from: g = 0.5 [ g1 + g2 + (g1 - g2) * K]
276  {
277  auto K = (I1 * (I1 + 1.0) - I2 * (I2 + 1.0)) / (It * (It + 1.0));
278  auto gt = mut / It;
279  auto g1 = mu1 / I1;
280  auto g2 = (g1 * (K + 1.0) - 2.0 * gt) / (K - 1.0);
281  auto mu2 = g2 * I2;
282  auto gl2 = (gl1 == 0) ? 1 : 0;
283  auto F1 = volotkaBW_F(mu1, I1, l1, gl1);
284  auto F2 = volotkaBW_F(mu2, I2, l2, gl2);
285  return [=](double r, double rN) {
286  return (0.5 / gt) * (g1 * F1(r, rN) + g2 * F2(r, rN) +
287  K * (g1 * F1(r, rN) - g2 * F2(r, rN)));
288  };
289  }
