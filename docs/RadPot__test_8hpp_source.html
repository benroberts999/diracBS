207 std::pair<std::string, double>
208 helper::compare_QED(const std::vector<QEDData> &QEDdata,
209  QED::RadPot::Scale scale) {
210  // Store worst comparison for unit test
211  double eps_worst = 0.0;
212  std::string at_worst;
213 
214  // Loop through atoms:
215  for (const auto &[atom, data] : QEDdata) {
216  std::cout << atom.name << ":\n";
217 
218  // Must use exact same rms value as above papers
219  const double rrms = atom.name == "E119" ? 6.5 : -1.0;
220 
221  // Construct wavefunction, solve HF core+valence (without QED):
222  Wavefunction wf0({5000, 1.0e-6, 120.0, 0.3 * 120.0, "loglinear", -1.0},
223  {atom.name, atom.A, "Fermi", rrms, 2.3}, 1.0);
224  wf0.hartreeFockCore("HartreeFock", 0.0, atom.core, 0.0, false);
225  wf0.hartreeFockValence(atom.val, false);
226 
227  // Construct wavefunction, solve HF core+valence (with QED):
228  Wavefunction wf({5000, 1.0e-6, 120.0, 0.3 * 120.0, "loglinear", -1.0},
229  {atom.name, atom.A, "Fermi", rrms, 2.3}, 1.0);
230  const auto rcut = 15.0;
231  wf.radiativePotential(scale, rcut, 1.0, {1.0}, false, false);
232  wf.hartreeFockCore("HartreeFock", 0.0, atom.core, 0.0, false);
233  wf.hartreeFockValence(atom.val, false);
234 
235  // Loop through each state (line in table) and compare data:
236  for (auto [state, t_de0, t_de] : data) {
237 
238  const auto Fv0 = wf0.getState(state);
239  const auto Fv = wf.getState(state);
240  if (!Fv0 || !Fv)
241  continue;
242 
243  // Operator (for first-order shift) - l-dependent
244  auto h = DiracOperator::Hrad(wf.qed->Vel(Fv->l()), wf.qed->Hmag(Fv->l()));
245 
246  // Calculate first-order Uehling shift (matrix element):
247  const auto de0 = h.radialIntegral(*Fv, *Fv);
248  // Find Hartree-Fock Uehling shift:
249  const auto de = Fv->en - Fv0->en;
250  // print results:
251  printf("%4s | %10.3e [%10.3e] | %10.3e [%10.3e]", state.c_str(), de0,
252  t_de0, de, t_de);
253  const auto eps0 = (de0 - t_de0) / (t_de0);
254  const auto eps = (de - t_de) / (t_de);
255  printf(" %6.0e %6.0e\n", eps0, eps);
256 
257  if (std::abs(eps) > eps_worst) {
258  eps_worst = std::abs(eps);
259  at_worst = atom.name + " " + state;
260  }
261  }
262  }
263  return {at_worst, eps_worst};
264 }
