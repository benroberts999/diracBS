144 inline double xCoshxMinusSinhx(double x) {
145  // x * std::cosh(x) - sinh(x) - large cancellation for low x
146  // Fix numerical instability by using expansion at low x

148  if (x < 0.5) {
149  const auto x3 = x * x * x;
