93  dVB_tdhf.solve_core(omega, max_iterations, print_details);
94  std::cout << "No Breit:\n";
95  dV_basis.solve_core(omega, max_iterations, print_details);
96  std::cout << "With Breit:\n";
97  dVB_basis.solve_core(omega, max_iterations, print_details);
98 
99  auto eps = 0.0;
100  std::string worst = "";
101 
102  std::cout << "dBr(%) TDHF TDHF(basis) " << h->name() << "\n";
103  for (const auto &Fv : wf.valence) {
104  for (const auto &Fw : wf.valence) {
105  if (Fw > Fv || h->isZero(Fv.k, Fw.k) || Fv.l() > max_l || Fw.l() > max_l)
106  continue;
107 
108  // lookup states in Breit wf (do this way, so no depend on order)
109  const auto &FvB = *wfB.getState(Fv.n, Fv.k);
110  const auto &FwB = *wfB.getState(Fw.n, Fw.k);
111 
112  const auto dv = dV_tdhf.dV(Fv, Fw);
113  const auto dvB = dVB_tdhf.dV(FvB, FwB);
114  const auto dv_basis = dV_basis.dV(Fv, Fw);
115  const auto dvB_basis = dVB_basis.dV(FvB, FwB);
116 
117  const auto dBr = (dvB - dv) / dv;
118  const auto dBr_basis = (dvB_basis - dv_basis) / dv_basis;
119  const auto str = Fv.shortSymbol() + "-" + Fw.shortSymbol();
120  printf("%7s %9.4e %9.4e\n", str.c_str(), dBr * 100.0, dBr_basis * 100.0);
121  const auto eps_this = std::abs((dBr - dBr_basis) / (dBr + dBr_basis));
122  if (eps_this > eps) {
123  eps = eps_this;
124  worst = str;
125  }
126  }
127  }
128  return {eps, worst};
129 }
