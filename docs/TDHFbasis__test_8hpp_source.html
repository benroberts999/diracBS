119 inline std::vector<std::pair<std::string, double>>
120 UnitTest::helper::dV_result_basis(const Wavefunction &wf,
121  const DiracOperator::TensorOperator &h,
122  double ww) {
123 
124  // Form TDHF (RPA) object for this operator
125  auto dV = ExternalField::TDHFbasis(&h, wf.getHF(), wf.basis);
126  // Solve set of TDHF equations for core, with frequency ww
127  const auto max_iterations = 150;
128  const auto print_details = true;
129  dV.solve_core(ww, max_iterations, print_details);
130 
131  std::vector<std::pair<std::string, double>> result;
132  for (const auto &Fv : wf.valence) {
133  for (const auto &Fm : wf.valence) {
134  if (h.isZero(Fv.k, Fm.k))
135  continue;
136 
137  result.emplace_back(Fv.shortSymbol() + Fm.shortSymbol(), dV.dV(Fv, Fm));
138  }
139  }
140  std::sort(begin(result), end(result),
141  [](auto a, auto b) { return a.second < b.second; });
142  return result;
143 }
